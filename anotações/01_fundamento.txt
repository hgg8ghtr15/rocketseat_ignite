--- tipo do solicitações
get = leitura
post = criar
put = atualiza
delete = deleta
pacth = atualiza parcial

---
1xx = information
2xx = sucesso

3xx = redirecionamento

4xx = erro
    400 = bad request
    401 = unauthorized
    403 = Forbidden
    404 = Not found
    422 = unprocessable entity

5xx = erro
    502 = bad gateway
